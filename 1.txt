    <script>
      let selectedType = "mp4";

      // Handle option selection
      document.querySelectorAll(".option-card").forEach((card) => {
        card.addEventListener("click", function () {
          document
            .querySelectorAll(".option-card")
            .forEach((c) => c.classList.remove("active"));
          this.classList.add("active");
          selectedType = this.dataset.type;

          // Clear info box when changing selection
          document.getElementById("infoBox").classList.remove("show");
        });
      });

      // Handle form submission
      document
        .getElementById("downloadForm")
        .addEventListener("submit", async function (e) {
          e.preventDefault();

          const videoUrl = document.getElementById("videoUrl").value.trim();
          const loadingEl = document.getElementById("loading");
          const errorEl = document.getElementById("errorMessage");
          const infoBoxEl = document.getElementById("infoBox");
          const downloadBtn = document.getElementById("downloadBtn");

          // Reset states
          errorEl.classList.remove("show");
          infoBoxEl.classList.remove("show");
          loadingEl.classList.add("show");
          downloadBtn.disabled = true;

          try {
            const baseUrl = "http://localhost:3000/api/youtube";

            if (selectedType === "info") {
              // Get video info
              const response = await fetch(
                `${baseUrl}/info?url=${encodeURIComponent(videoUrl)}`,
              );

              if (!response.ok) {
                throw new Error("Gagal mengambil informasi video");
              }

              const data = await response.json();

              // Display info
              infoBoxEl.innerHTML = `
                        ${data.thumbnail ? `<img src="${data.thumbnail}" alt="Thumbnail">` : ""}
                        <h3>${data.title || "Tidak ada judul"}</h3>
                        <p><strong>Durasi:</strong> ${data.duration || "N/A"}</p>
                        <p><strong>Channel:</strong> ${data.author || "N/A"}</p>
                        <p><strong>Views:</strong> ${data.views || "N/A"}</p>
                    `;
              infoBoxEl.classList.add("show");
            } else {
              // Download video or audio
              const endpoint =
                selectedType === "mp4" ? "download" : "download-mp3";
              const downloadUrl = `${baseUrl}/${endpoint}?url=${encodeURIComponent(videoUrl)}`;

              // Create hidden link and trigger download
              const link = document.createElement("a");
              link.href = downloadUrl;
              link.download = "";
              document.body.appendChild(link);
              link.click();
              document.body.removeChild(link);

              // Show success message
              infoBoxEl.innerHTML = `
                        <h3>✅ Download Dimulai!</h3>
                        <p>File ${selectedType.toUpperCase()} Anda sedang diunduh...</p>
                    `;
              infoBoxEl.classList.add("show");
            }
          } catch (error) {
            errorEl.textContent =
              "❌ " +
              (error.message ||
                "Terjadi kesalahan. Pastikan URL valid dan server berjalan.");
            errorEl.classList.add("show");
          } finally {
            loadingEl.classList.remove("show");
            downloadBtn.disabled = false;
          }
        });

      // Auto-focus on input when page loads
      window.addEventListener("load", () => {
        document.getElementById("videoUrl").focus();
      });
    </script>
